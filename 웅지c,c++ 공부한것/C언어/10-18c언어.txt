#include <stdio.h>

int main(void)
{
	int total=0, num=0;

	do //do while문은 최소 1번실행
	{
		printf("정수 입력(0입력시 빠져나감): ");
		scanf("%d",&num);
		total += num; //total = total + num
	}while(num!=0); //num이 0이아니면.
	printf("합계 : %d \n",total);
	return 0;
}
