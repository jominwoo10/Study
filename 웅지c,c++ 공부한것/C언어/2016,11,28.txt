#include <stdio.h>

int main(void)
{
	int minwoo[3]={10,20,30};
	int * jo=minwoo;  //int * jo=&minwoo[0]과 같다
	
	printf("%d %d %d %d \n",*jo, *(jo+1), *(jo+2), *jo+1);
	printf("%d ",*jo); jo++;
	printf("%d ",*jo); jo++;
	printf("%d ",*jo); jo--;
	printf("%d \n",*jo);
	return 0;
}
