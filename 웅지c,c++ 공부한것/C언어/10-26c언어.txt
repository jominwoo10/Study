{

	scanf("%d",&num);

	if(num==1)
		goto ONE;  //1이 입력되면 goto에있는 ONE를 만나 밑에 ONE:줄에가서 실행
	else if(num==2)
		goto TWO;
	else if(num==3)
		goto THREE;
	else
		goto OTHER;

ONE:

OTHER:

END:
	return 0;

-switch를 사용하여 계산기 만들기-

#include <stdio.h>

int main(void)
{	
	int num;
	double num2;
	double num3;
	double result;

	printf("두개의 정수 입력 : ");
	scanf("%lf %lf",&num2,&num3);
	printf("계산기를 이용할 연산자 입력 \n");
	printf("1.덧셈 2뺄셈 3곱셈 4나눗셈 : ");
	scanf("%d",&num);

	switch(num)
	{
	case 1:
		result = num2 + num3;
		printf("%f + %f = %f",num2,num3,result);
		break;
	case 2:
		result = num2 - num3;
		printf("%f - %f = %f",num2,num3,result);
		break;
	case 3:
		result = num2 * num3;
		printf("%f * %f = %f",num2,num3,result);
		break;
	case 4:
		result = num2 / num3;
		printf("%f / %f = %f",num2,num3,result);
		break;
	default:
		printf("계산기 사용 불가");
	}

	return 0;
}

-함수를 이용한 계산기 만들기-

#include <stdio.h>

int add(int num1,int num2)  //add 함수 선언(쳐먹는것 o 똥싸는것 o)
{
	return num1+num2;
}

int minus(int num1,int num2)
{
	return num1-num2;
}

int mul(int num1,int num2)
{
	return num1*num2;
}

int div(int num1,int num2)
{
	return num1/num2;
}
int main(void)
{	
	int num1,num2;
	char d;
	printf("계산할 두개의 정수와 연산자를 입력해주세요 \n");
	printf("연산자는 + - * / 로 입력해주세요 \n");
	scanf("%d %c %d",&num1,&d,&num2);

	if(d == '+')
		printf("%d + %d = %d \n",num1,num2,add(num1,num2));
	else if(d == '-')
		printf("%d - %d = %d \n",num1,num2,minus(num1,num2));
	else if(d == '*')
		printf("%d * %d = %d \n",num1,num2,mul(num1,num2));
	else if(d == '/')
		printf("%d / %d = %d \n",num1,num2,div(num1,num2));
	else
		printf("계산할수 없습니다.");
	return 0;
}
