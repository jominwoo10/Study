#include <stdio.h>

int compare(int num1, int num2);
int absovalue(int num);

int main(void)
{
	int num1, num2;
	printf("두 정수 입력 : ");
	scanf("%d %d",&num1,&num2);
	printf("%d와 %d중 절댓값이 큰 정수 : %d \n",num1,num2,compare(num1,num2));
	return 0;
}

int compare(int num1, int num2)
{
	if(absovalue(num1) > absovalue(num2))  //함수의 함수를 선언할수도 있다.
		return num1;
	else
		return num2;
}

int absovalue(int num)
{
	if(num<0)
		return num*(-1);
	else
		return num;
}

	return 0;
}

int funcone(void)
{
	int num=10;
	num++;
	printf("funcone num : %d \n",num);  //후위증가로인해 11출력
	return 0;
}

int functwo(void)
{
	int num1=20;
	int num2=30;
	num1++, num2--;
	printf("num1 & num2 : %d %d \n",num1,num2);  //위에서 후위증가와 후위감소로인해 21과 29출력
	return 0;
}
