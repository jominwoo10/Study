#include <stdio.h>

int main(void)
{
	int total=0;
	int i, num;
	printf("0부터 num까지의 덧셈 결과, num은? : ");
	scanf("%d",&num);

	for(i=0; i<num+1; i++) //i는 0부터 num+i보다작을때까지 1씩증가
		total+=i; //total = total+i

	printf("0부터 %d까지 덧셈의 결과 : %d \n",num,total);
	return 0;
}

#include <stdio.h>

int main(void)
{
	int opt;
	double num1, num2;
	double result;
	
	printf("1.더하기 2.빼기 3.곱하기 4.나누기 \n");  //계산기
	printf("번호 선택 : ");
	scanf("%d",&opt);
	printf("두개의 실수 입력 : ");
	scanf("%lf %lf,&num1,&num2);

	if(opt==1)
		result = num1 + num2;  //opt가 1이면 덧셈 실행
	if(opt==2)
		result = num1 - num2;  //opt가 2이면 뺄셈 실행
	if(opt==3)
		result = num1 * num2;  //opt가 3이면 곱셈 실행
	if(opt==4)
		result = num1 / num2;  //opt가 4이면 나눗셈 실행

	printf("결과 : %f \n",result);
	return 0;
}
