#include <stdio.h>

int main(void)
{
	char ch=11;
	int inum=1824;
	double dnum=3.356;
	printf("ch의 크기 : %d \n",sizeof(ch)); //sizeof는 자료형의 크기를알수있다 ch는 char형이므로 1byte라서 1값출력
	printf("inum의 크기 : %d \n",sizeof(inum)); //inum은 int형이므로 4byte라서 4값출력
	printf("dnum의 크기 : %d \n",sizeof(dnum)); //dnum은 double형이므로 8byte라서 8값출력

	printf("char의 크기 : %d \n",sizeof(char)); //char = 1byte
	printf("int의 크기 : %d \n",sizeof(int)); //int = 4byte
	printf("long의 크기 : %d \n",sizeof(long)); //long = 4byte
	printf("long long의 크기 : %d \n",sizeof(long long)); //long long = 8byte
	printf("float의 크기 : %d \n",sizeof(float)); //float = 4byte
	printf("double의 크기 : %d \n",sizeof(double)); //double = 8byte
	return 0;
}

#include <stdio.h>

int main(void)
{
	char num1=1, num2=2, result1=0;
	short num3=500, num4=600, result2=0;

	printf("num1의 자료형 값과 num2의 자료형 값 : %d, %d \n",sizeof(num1),sizeof(num2)); //char 자료형값 1,1출력
	printf("num3의 자료형 값과 num4의 자료형 값 : %d, %d \n",sizeof(num3),sizeof(num4)); //short 자료형값 2,2출력

	printf("char형끼리 더한 값 : %d \n",sizeof(num1+num2)); //연산이되면서 int형으로 넘어가서 4출력
	printf("short형끼리 더한 값 : %d \n",sizeof(num3+num4)); //연산이되면서 int형으로 넘어가서 4출력

	result1=num1+num2;
	result2=num3+num4;
	printf("결과1 자료형값과 결과2 자료형값 : %d,%d \n",sizeof(result1),sizeof(result2));

	return 0;
}

#include <stdio.h>

int main(void)
{
	double rad;
	double area;
	printf("원의 반지름 입력: ");
	scanf("%lf", &rad); //입력할때는 %lf를써줘야한다 = double

	area = rad*rad*3.1415;
	printf("원의 넓이 : %f \n",area); //출력할때는 %f입력
